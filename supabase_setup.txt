-- ============================================================
-- PATIENT MANAGEMENT SYSTEM - SUPABASE ONE-TIME SQL SETUP
-- ============================================================
-- Run this ONCE in your Supabase SQL Editor (Dashboard > SQL Editor > New Query)
-- This creates all 3 tables, enables RLS, and sets open access policies.
-- ============================================================

-- 1. PATIENTS TABLE
CREATE TABLE IF NOT EXISTS patients (
  id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  reg_no TEXT UNIQUE NOT NULL,
  name TEXT NOT NULL,
  screening_date TEXT,
  age TEXT,
  sex TEXT,
  residence TEXT,
  height TEXT,
  duration_mc TEXT,
  co_morbidities TEXT,
  risk_factors TEXT,
  treatment_taken TEXT,
  weight_kg TEXT,
  education_status TEXT,
  socio_economic_status TEXT,
  bmi TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. BASELINE TREATMENTS TABLE
CREATE TABLE IF NOT EXISTS baseline_treatments (
  id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  patient_id BIGINT NOT NULL REFERENCES patients(id) ON DELETE CASCADE,
  date TEXT,
  present_complaint TEXT,
  medicine_prescribed TEXT,
  potency TEXT,
  dose_doses TEXT,
  miasm_data JSONB DEFAULT '{}',
  susceptibility_data JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Only one baseline per patient
CREATE UNIQUE INDEX IF NOT EXISTS idx_baseline_patient
  ON baseline_treatments(patient_id);

-- 3. OUTCOME ASSESSMENTS TABLE
CREATE TABLE IF NOT EXISTS outcome_assessments (
  id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  patient_id BIGINT NOT NULL REFERENCES patients(id) ON DELETE CASCADE,
  assessment_number INTEGER NOT NULL CHECK (assessment_number BETWEEN 1 AND 6),
  date TEXT NOT NULL,
  brief_notes TEXT,
  medicine_prescribed TEXT,
  potency TEXT,
  dose_doses TEXT,
  oridl_main_complaint TEXT,
  oridl_wellbeing TEXT,
  prathot_version TEXT,
  miasm_data JSONB,
  susceptibility_data JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- One assessment per number per patient
CREATE UNIQUE INDEX IF NOT EXISTS idx_outcome_patient_num
  ON outcome_assessments(patient_id, assessment_number);

-- ============================================================
-- ROW LEVEL SECURITY (RLS) - Open access via anon key
-- ============================================================

ALTER TABLE patients ENABLE ROW LEVEL SECURITY;
ALTER TABLE baseline_treatments ENABLE ROW LEVEL SECURITY;
ALTER TABLE outcome_assessments ENABLE ROW LEVEL SECURITY;

-- Patients: full access
CREATE POLICY "patients_select" ON patients FOR SELECT USING (true);
CREATE POLICY "patients_insert" ON patients FOR INSERT WITH CHECK (true);
CREATE POLICY "patients_update" ON patients FOR UPDATE USING (true) WITH CHECK (true);
CREATE POLICY "patients_delete" ON patients FOR DELETE USING (true);

-- Baseline treatments: full access
CREATE POLICY "baseline_select" ON baseline_treatments FOR SELECT USING (true);
CREATE POLICY "baseline_insert" ON baseline_treatments FOR INSERT WITH CHECK (true);
CREATE POLICY "baseline_update" ON baseline_treatments FOR UPDATE USING (true) WITH CHECK (true);
CREATE POLICY "baseline_delete" ON baseline_treatments FOR DELETE USING (true);

-- Outcome assessments: full access
CREATE POLICY "outcome_select" ON outcome_assessments FOR SELECT USING (true);
CREATE POLICY "outcome_insert" ON outcome_assessments FOR INSERT WITH CHECK (true);
CREATE POLICY "outcome_update" ON outcome_assessments FOR UPDATE USING (true) WITH CHECK (true);
CREATE POLICY "outcome_delete" ON outcome_assessments FOR DELETE USING (true);

-- ============================================================
-- DONE! All tables and policies are ready.
-- ============================================================
